CONSTRUIR IMAGEN SERVER (EN LA RUTA 'server/'): docker build -t servidor_http .
CONSTRUIR CONTENEDOR SERVER: docker run -d --name tallerderedes_server servidor_http

CONSTRUIR IMAGEN CLIENTE (EN LA RUTA 'client/'): docker build -t cliente_http .
CONSTRUIR CONTENEDOR CLIENTE: docker run -d -it --name tallerderedes_cliente cliente_http

EN UN POWERSHELL EMPEZAR CAPTURA DE TRAFICO: docker exec -it tallerderedes_cliente tcpdump -w /app/captura.pcap
HACER PETICION CORRECTA EN BASH DE CLIENTE: docker exec tallerderedes_cliente python3 consulta.py 172.17.0.2
HACER PETICION FALLIDA EN BASH DE CLIENTE: docker exec tallerderedes_cliente python3 consulta_falla.py 172.17.0.2
EN EL POWERSHELL PARAR CAPTURA: CONTROL + C
EN EL POWERSHELL GUARDAR CAPTURA LOCALMENTE (EN LA RUTA DEL POWERSHELL): docker cp tallerderedes_cliente:/app/captura.pcap .
